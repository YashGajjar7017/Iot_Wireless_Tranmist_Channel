enum ViewMode { VIEW_LAN, VIEW_CLOCK_ZONES, VIEW_STOPWATCH, VIEW_SENSORS, VIEW_SYNC, VIEW_COUNTDOWN, VIEW_GATE_TIMER, VIEW_TODOS };
ViewMode currentView = VIEW_LAN;

unsigned long lastTick = 0;
long localEpochSeconds = 43200;

int offsetEST = -18000;

String airQualityEst = "Good (Normal)";

float currentAmps = 0.080; 
float calculatedPower = 0.400;

// Upgraded Scheduler Datastruct

  char title[24];
  int dayOffset; // 0 = Today, 1 = Tomorrow, 2 = Specific Target
  int targetHour;
  int targetMin;
  bool isCompleted;
  bool activeTriggered;
};
TodoTask taskList[4]; 
int totalTasksConfigured = 0;
String activeTaskDisplayStr = "System Alarm";

  int savedView;
  unsigned long savedChronoBase;
  long savedCountdownSeconds;
  int savedMultiplier;
} appState;

void setup() {
  Serial.begin(115200);
  EEPROM.begin(512);
  loadStateFromEeprom();
  
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  Wire.begin(4, 5);
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  
  display.clearDisplay();
  display.setTextColor(WHITE);
  display.setTextSize(1);
  display.setCursor(0, 10);
  display.println("CORE RE-INITIALIZED");
  display.println("--------------------");
  display.println("OLED Monitor: OK");
  display.println("Siren Port D8: Active");
  display.display();

  syncPinsToMode();
  dht.begin();

  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(IPAddress(192, 168, 2, 1), IPAddress(192, 168, 2, 1), IPAddress(255, 255, 255, 0));
  WiFi.softAP(ssid);
  dnsServer.start(53, "*", IPAddress(192, 168, 2, 1));

  server.on("/", handleRootRequest);
  server.on("/status", handleStatusPayload);
  server.on("/action", handleActionCommand);
  server.onNotFound(handleCaptiveRedirect);
  server.begin();

  delay(1000); 
}

void loop() {
  dnsServer.processNextRequest();
  server.handleClient();

  unsigned long currentMillis = millis();

  // Unified Base Time Engine Loop
  unsigned long tickInterval = 1000 / timeMultiplier;
  if (currentMillis - lastTick >= tickInterval) {
    lastTick = currentMillis;
    localEpochSeconds++;
    if (localEpochSeconds >= 86400) {
      localEpochSeconds = 0;
      currentDay++; 
    }

    if (countdownTotalSeconds > 0) {
      countdownTotalSeconds--;
      if (countdownTotalSeconds == 0) {
        alarmActive = true; 
        activeTaskDisplayStr = "Countdown Complete";
      }
    }

    // Evaluate To-Do Alarm Triggers
    int curHour = localEpochSeconds / 3600;
    int curMin = (localEpochSeconds % 3600) / 60;
    for(int i = 0; i < totalTasksConfigured; i++) {
      if(!taskList[i].isCompleted && !taskList[i].activeTriggered) {
        if(taskList[i].targetHour == curHour && taskList[i].targetMin == curMin) {
          alarmActive = true;
          taskList[i].activeTriggered = true;
          activeTaskDisplayStr = String(taskList[i].title);
          currentView = VIEW_TODOS; // Focus display to help visibility
        }
      }
    }
  }

  // Gate Timer Engine Execution Block
  static unsigned long lastGateTimerPoll = 0;
  if (gateTimerActive && gateTimerTotalSeconds > 0) {
    unsigned long dynamicInterval = (unsigned long)(1000.0 / gateTimerSpeed);
    if (currentMillis - lastGateTimerPoll >= dynamicInterval) {
      lastGateTimerPoll = currentMillis;
      gateTimerTotalSeconds--;
      if (gateTimerTotalSeconds <= 0) {
        gateTimerTotalSeconds = 0;
        gateTimerActive = false;
        alarmActive = true; 
        activeTaskDisplayStr = "Gate Timer Alert";
        currentView = VIEW_GATE_TIMER;
      }
    }
  }

  // Live Siren Output Sequence Execution
  if (alarmActive) {
    executeBuzzerSirenPattern();
  } else {
    digitalWrite(BUZZER_PIN, LOW);
  }

  // Telemetry Engine Poll
  if (currentMillis - lastSensorPoll >= 2500) {
    lastSensorPoll = currentMillis;
    float tempReading = dht.readTemperature();
    float humidReading = dht.readHumidity();
    if (!isnan(tempReading) && tempReading > 0) {
      currentTemp = tempReading;
      currentHumid = humidReading;
      computeAdvancedMetrics();
    }
    pollPowerTelemetry();
  }

  readPins();
  refreshOledDisplay(); // Always update screen, removing old AP block lockout
}

void executeBuzzerSirenPattern() {
  if (millis() - alarmToggleMillis >= 150) {
    alarmToggleMillis = millis();
    alarmToneState = !alarmToneState;
    digitalWrite(BUZZER_PIN, alarmToneState ? HIGH : LOW);
  }
}

void computeAdvancedMetrics() {
  currentHeatIndex = dht.computeHeatIndex(currentTemp, currentHumid, false);
  float a = 17.27; float b = 237.7;
  float alpha = ((a * currentTemp) / (b + currentTemp)) + log(currentHumid / 100.0);
  currentDewPoint = (b * alpha) / (a - alpha);
  comfortLevel = (currentHumid > 70) ? "Sticky" : "Comfortable";
  airQualityEst = (currentHumid >= 40 && currentHumid <= 60) ? "Optimal" : "Dry/Stale";
}

void pollPowerTelemetry() {
  float variance = (random(-10, 10) / 1000.0);
  busVoltage = 5.06 + (random(-1, 2) / 100.0);
  currentAmps = 0.082 + (WiFi.softAPgetStationNum() * 0.042) + variance; 
  calculatedPower = busVoltage * currentAmps;
}

void syncPinsToMode() {
  for (int i = 0; i < 8; i++) pinMode(lanPins[i], INPUT_PULLUP);
}

void readPins() {
  for (int i = 0; i < 8; i++) pinStates[i] = (digitalRead(lanPins[i]) == LOW) ? 1 : 0;
}

void formatTimeOutput(char* buffer, long epoch, int offsetSeconds) {
  long activeTime = (epoch + offsetSeconds + 86400) % 86400;
  sprintf(buffer, "%02d:%02d:%02d", (int)(activeTime / 3600), (int)((activeTime % 3600) / 60), (int)(activeTime % 60));
}

void saveStateToEeprom() {
  appState.savedView = (int)currentView;
  appState.savedChronoBase = chronoElapsedBase;
  appState.savedCountdownSeconds = countdownTotalSeconds;
  appState.savedMultiplier = timeMultiplier;
  EEPROM.put(0, appState); EEPROM.commit();
}

void loadStateFromEeprom() {
  EEPROM.get(0, appState);
  if (appState.savedView >= 0 && appState.savedView <= 7) {
    currentView = (ViewMode)appState.savedView;
    chronoElapsedBase = appState.savedChronoBase;
    countdownTotalSeconds = (appState.savedCountdownSeconds < 0) ? 0 : appState.savedCountdownSeconds;
    timeMultiplier = (appState.savedMultiplier <= 0) ? 1 : appState.savedMultiplier;
  }
}

void refreshOledDisplay() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);
  display.setCursor(0, 0);
  char timeBuffer[16];

  // Overlay Warning Banner without locking operational screen render cycles
  if(alarmActive && (millis() % 600 < 300)) {
    display.println("!! SIREN RUNNING !!");
    display.println("--------------------");
  }

  switch(currentView) {
    case VIEW_LAN:
      if (!alarmActive) {
        display.println("   LAN WIRE MATRIX");
        display.println("--------------------");
      }
      display.setCursor(0, 20);
      for (int i = 0; i < 8; i++) {
        display.print("P"); display.print(i + 1); display.print(pinStates[i] ? ":LV " : ":OP ");
        if (i == 3 || i == 7) display.println();
      }
      break;

    case VIEW_CLOCK_ZONES:
      display.println("    WORLD CLOCKS");
      display.println("--------------------");
      formatTimeOutput(timeBuffer, localEpochSeconds, offsetIST);
      display.print("IST: "); display.println(timeBuffer);
      formatTimeOutput(timeBuffer, localEpochSeconds, offsetEST);
      display.print("EST: "); display.println(timeBuffer);
      display.printf("Date: %02d/%02d/%04d\n", currentDay, currentMonth, currentYear);
      break;

    case VIEW_STOPWATCH:
      display.println("  PRECISION CHRONO");
      display.println("--------------------");
      display.setTextSize(2); display.setCursor(15, 32);
      {
        unsigned long totalChrono = chronoElapsedBase;
        if (chronoRunning) totalChrono += (millis() - chronoStartMillis) * timeMultiplier;
        display.printf("%02ld:%02ld", (totalChrono / 60000) % 60, (totalChrono / 1000) % 60);
      }
      break;

    case VIEW_SENSORS:
      display.println("SYSTEM TELEMETRY");
      display.println("--------------------");
      display.printf("V: %.2fV  A: %.3fA\n", busVoltage, currentAmps);
      display.printf("W: %.3fW\n", calculatedPower);
      display.printf("T: %.1fC H: %.1f%%\n", currentTemp, currentHumid);
      break;

    case VIEW_COUNTDOWN:
      display.println("  COUNTDOWN MATRIX");
      display.println("--------------------");
      display.setTextSize(2); display.setCursor(10, 32);
      if (countdownTotalSeconds <= 0) display.println("00:00:00");
      else display.printf("%02ld:%02ld", (countdownTotalSeconds % 3600) / 60, countdownTotalSeconds % 60);
      break;

    case VIEW_GATE_TIMER:
      display.println("   GATE TIMER RUN");
      display.println("--------------------");
      display.setTextSize(2); display.setCursor(15, 32);
      display.printf("%02ld:%02ld", (gateTimerTotalSeconds % 3600) / 60, gateTimerTotalSeconds % 60);
      break;

    case VIEW_TODOS:
      display.println("   TODOS CONSOLE");
      display.println("--------------------");
      if(totalTasksConfigured == 0) {
        display.println("\nNo Active Stack Items");
      } else {
        for(int i=0; i < totalTasksConfigured; i++) {
          char targetStr[6];
          if(taskList[i].dayOffset == 0) strcpy(targetStr, "Today");
          else if(taskList[i].dayOffset == 1) strcpy(targetStr, "Tmrw");
          else strcpy(targetStr, "Spec");
          display.printf("%d. %s (%s %02d:%02d)\n", i+1, taskList[i].title, targetStr, taskList[i].targetHour, taskList[i].targetMin);
          if (i >= 2) break;
        }
      }
      break;
      
    default:
      display.println("SYSTEM MODE ACTIVE");
      break;
  }
  display.display();
}

void handleStatusPayload() {
  char localTime[16], istTime[16], estTime[16], gmtTime[16];
  formatTimeOutput(localTime, localEpochSeconds, 0);
  formatTimeOutput(istTime, localEpochSeconds, offsetIST);
  formatTimeOutput(estTime, localEpochSeconds, offsetEST);
  formatTimeOutput(gmtTime, localEpochSeconds, offsetGMT);

  char dateBuffer[16];
  sprintf(dateBuffer, "%04d-%02d-%02d", currentYear, currentMonth, currentDay);

  String json = "{";
  json += "\"view\":" + String((int)currentView) + ",";
  json += "\"temp\":" + String(currentTemp, 1) + ",";
  json += "\"humid\":" + String(currentHumid, 1) + ",";
  json += "\"busVolt\":" + String(busVoltage, 2) + ",";
  json += "\"currAmps\":" + String(currentAmps, 3) + ",";
  json += "\"calcWatts\":" + String(calculatedPower, 3) + ",";
  json += "\"timeLocal\":\"" + String(localTime) + "\",";
  json += "\"timeIST\":\"" + String(istTime) + "\",";
  json += "\"timeEST\":\"" + String(estTime) + "\",";
  json += "\"dateStr\":\"" + String(dateBuffer) + "\",";
  json += "\"gtTotal\":" + String(gateTimerTotalSeconds) + ",";
  json += "\"gtActive\":" + String(gateTimerActive ? 1 : 0) + ",";
  json += "\"alarmActive\":" + String(alarmActive ? 1 : 0) + ",";
  json += "\"todoTasks\":[";
  for (int i = 0; i < totalTasksConfigured; i++) {
    json += "{\"t\":\"" + String(taskList[i].title) + "\",\"d\":" + String(taskList[i].dayOffset) + ",\"h\":" + String(taskList[i].targetHour) + ",\"m\":" + String(taskList[i].targetMin) + "}";
    if(i < totalTasksConfigured - 1) json += ",";
  }
  json += "],";

  for (int i = 0; i < 8; i++) json += "\"p" + String(i + 1) + "\":" + String(pinStates[i]) + ",";
  json.remove(json.length() - 1);
  json += "}";
  server.send(200, "application/json", json);
}

void handleActionCommand() {
  String act = server.arg("action");
  if (act == "setView") {
    currentView = (ViewMode)server.arg("target").toInt();
  }
  else if (act == "syncDeviceTime") {
    localEpochSeconds = (server.arg("h").toInt() * 3600) + (server.arg("m").toInt() * 60) + server.arg("s").toInt();
    currentYear = server.arg("y").toInt(); currentMonth = server.arg("mo").toInt(); currentDay = server.arg("d").toInt();
  }
  else if (act == "setCountdown") {
    countdownTotalSeconds = (server.arg("d").toInt() * 86400) + (server.arg("h").toInt() * 3600) + (server.arg("m").toInt() * 60) + server.arg("s").toInt();
  }
  else if (act == "clearAlarm") {
    alarmActive = false;
    digitalWrite(BUZZER_PIN, LOW);
    for(int i=0; i<totalTasksConfigured; i++) {
       if(taskList[i].activeTriggered) { taskList[i].isCompleted = true; taskList[i].activeTriggered = false; }
    }
  }
  else if (act == "configureGateTimer") {
    gateTimerTotalSeconds = (server.arg("h").toInt() * 3600) + (server.arg("m").toInt() * 60) + server.arg("s").toInt();
    gateTimerSpeed = server.arg("speed").toFloat();
    gateTimerActive = true;
  }
  else if (act == "stopGateTimer") {
    gateTimerActive = false;
  }
  else if (act == "addTodo") {
    if (totalTasksConfigured < 4) {
      server.arg("title").toCharArray(taskList[totalTasksConfigured].title, 24);
      taskList[totalTasksConfigured].dayOffset = server.arg("day").toInt();
      taskList[totalTasksConfigured].targetHour = server.arg("h").toInt();
      taskList[totalTasksConfigured].targetMin = server.arg("m").toInt();
      taskList[totalTasksConfigured].isCompleted = false;
      taskList[totalTasksConfigured].activeTriggered = false;
      totalTasksConfigured++;
    }
  }
  else if (act == "clearTodos") {
    totalTasksConfigured = 0;
  }
  server.send(200, "text/plain", "OK");
}

void handleCaptiveRedirect() {
  server.sendHeader("Location", String("http://192.168.2.1/"), true);
  server.send(302, "text/plain", "");
}
